#include <iostream>
using namespace std;

int& increment(int& num) {
    num++;
    return num;
}

int main() {
    int x = 5;
    int& y = increment(x);
    std::cout << x << std::endl; // 输出 6
    y = 10;
    std::cout << x << std::endl; // 输出 10
    return 0;
}
