    #include <iostream>
using namespace std;

class Car {
   public:
    int m_price;
    Car(int price = 9) : m_price(price) {}
};

Car getTemp() { return Car(); }

int main(int argv, char* argc[]) {
    int&& a = 3;  // 3 是右值，a 是右值引用，但它是一个左值（能取地址）

    int b = 4;  // b 和 4 都是左值

    int&& c = b + 1;        // b+5 是右值，c 是右值引用（同时也是左值）
    Car&& car = getTemp();  // getTemp() 返回的是右值（临时变量）,car 是右值引用（同时也是左值）

    cout << "a=" << a << endl;
    cout << "c=" << c << endl;
    cout << "car.m_price=" << car.m_price << endl;

    // 下面3个变量都可以取地址，是左值
    cout << "&a=" << &a << endl;
    cout << "&c=" << &c << endl;
    cout << "&car.m_price=" << &car.m_price << endl;
    return 0;
}

移动语义

如果一个对象中有堆区资源，需要编写拷贝构造函数和赋值函数，实现深拷贝

深拷贝把对象中的堆区资源赋值一份，如果源对象是临时对象，拷贝完就释放了，这样会造成没有意义的资源申请与资源释放，如果能够直接使用源对象拥有的资源，可以节省资源的申请和释放的时间，C++11新增加的移动语义就能够做到

拷贝构造函数和赋值函数

拷贝构造函数和赋值函数都是用于对象拷贝的，但它们有不同的调用时机和语义：

  1. 拷贝构造函数是在创建新对象并用另一个同类型对象初始化它时调用的。例如：
1
2

MyClass a;
MyClass b(a); // 调用拷贝构造函数，将 b 初始化为 a 的副本

拷贝构造函数的声明通常为：

1

MyClass(const MyClass& other);

其中 other 是要拷贝的对象的引用。

赋值函数是在已有对象被赋值为另一个同类型对象时调用的。例如：

1
2

MyClass a, b;
b = a; // 调用赋值函数，将 b 赋值为 a 的副本

赋值函数的声明通常为：

1

MyClass& operator=(const MyClass& other);

其中 other 是要拷贝的对象的引用，返回值是当前对象的引用。

需要注意的是，C++11 引入了移动构造函数和移动赋值函数，用于处理右值引用（rvalue reference）的情况。这两个函数在效率上有很大提升，但语义和使用方式与拷贝构造函数和赋值函数有所不同。

class MyClass {
public:
    // 构造函数
    MyClass() : ptr(nullptr) {}

    // 拷贝构造函数
    MyClass(const MyClass& other) {
        if (other.ptr != nullptr) {
            ptr = new int(*other.ptr);
        }
    }

    // 赋值函数
    MyClass& operator=(const MyClass& other) {
        if (this != &other) {
            // 释放当前对象的资源
            delete ptr;
            // 分配新的资源
            if (other.ptr != nullptr) {
                ptr = new int(*other.ptr);
            } else {
                ptr = nullptr;
            }
        }
        return *this;
    }

    // 析构函数
    ~MyClass() {
        delete ptr;
    }

private:
    int* ptr;
};

在赋值函数中，需要首先判断是否为自我赋值，若是则直接返回当前对象的引用；否则需要先释放当前对象所持有的资源，再根据other对象是否持有资源进行不同的处理，最后返回当前对象的引用。这里返回左值引用是为了支持链式赋值操作，例如a = b = c
