openRequest.onsuccess = function() {
  let db = openRequest.result;
  // 继续使用 db 对象处理数据库
};

onupgradeneeded、onerror、onsuccess这几个属性有点像回调函数，当indexedDB.open("store", 1);打开数据库成功调用 onsuccess 函数，打开失败调用 onerror 函数

其中onupgradeneeded属性可以写创建数据库以及升级数据库版本的初始化或者升级代码操作

对象库

引用自：https://zh.javascript.info/indexeddb#dui-xiang-ku-objectstore

要在 IndexedDB 中存储某些内容，我们需要一个 对象库。

对象库是 IndexedDB 的核心概念，在其他数据库中对应的对象称为“表”或“集合”。它是储存数据的地方。一个数据库可能有多个存储区：一个用于存储用户数据，另一个用于商品，等等。

尽管被命名为“对象库”，但也可以存储原始类型。

几乎可以存储任何值，包括复杂的对象。

IndexedDB 使用 标准序列化算法 来克隆和存储对象。类似于 JSON.stringify，不过功能更加强大，能够存储更多的数据类型。

有一种对象不能被存储：循环引用的对象。此类对象不可序列化，也不能进行 JSON.stringify。

库中的每个值都必须有唯一的键 key。

键的类型必须为数字、日期、字符串、二进制或数组。它是唯一的标识符，所以我们可以通过键来搜索/删除/更新值。

/**
 * 打开或创建指定名称和版本的 IndexedDB 数据库
 * @param {string} dbName - 数据库名称
 * @param {string} storeName - 对象存储的名称
 * @param {Object} keyOptions - 键选项对象，包括 keyPath 和 autoIncrement
 * @param {number} version - 数据库版本（默认为1）
 * @returns {Promise<IDBDatabase>} - 返回一个 Promise，解析为打开或创建的 IndexedDB 数据库实例
 */
const openDB = options => {
  const { dbName, storeName, keyOptions, version = 1 } = options

  return new Promise((resolve, reject) => {
    // 兼容不同浏览器的 IndexedDB 实现
    var indexedDB =
      window.indexedDB ||
      window.mozIndexedDB ||
      window.webkitIndexedDB ||
      window.msIndexedDB

    let db

    // 创建或打开数据库的请求
    let openRequest = indexedDB.open(dbName, version)

    // 成功打开数据库时的回调
    openRequest.onsuccess = () => {
      // 获取数据库实例
      db = openRequest.result
      console.log(`${dbName}数据库打开成功`)
      // 将数据库实例返回出去
      resolve(db)
    }

    // 打开数据库失败时的回调
    openRequest.onerror = () => {
      console.log(`${dbName}数据库打开失败`)
      reject(`${dbName}数据库打开失败`)
    }

    // 数据库版本升级时的回调，用于创建或修改数据库结构
    openRequest.onupgradeneeded = () => {
      console.log("onupgradeneeded")
      db = openRequest.result

      // 判断是否已存在名为 "part" 的对象存储
      if (!db.objectStoreNames.contains(storeName)) {
        console.log(storeName)
        // 如果不存在，则创建
        // 创建对象存储，根据传递的参数设置选项对象
        const storeOptions = {}
        if (keyOptions) {
          if (keyOptions.keyPath) {
            storeOptions.keyPath = keyOptions.keyPath
          }
          if (keyOptions.autoIncrement) {
            storeOptions.autoIncrement = true
          }
        }
        // 创建对象存储
        db.createObjectStore(storeName, storeOptions)
      }
    }
  })
}

/**
 * 向 IndexedDB 中指定的对象存储添加数据
 * @param {IDBDatabase} db - IndexedDB 数据库实例
 * @param {string} storeName - 对象存储的名称
 * @param {any} data - 要添加的数据
 */
const addData = (db, storeName, data) => {
  // 启动事务，指定对象存储和操作模式为读写
  const transaction = db.transaction([storeName], "readwrite")

  // 获取对象存储进行操作
  const store = transaction.objectStore(storeName)

  // 添加数据到对象存储
  // 判断 data 是不是一个数组对象
  let addRequest
  if (Array.isArray(data)) {
    data.forEach(item => {
      addRequest = store.add(item)
    })
  } else {
    addRequest = store.add(data)
  }

  // 事务成功时触发的事件
  transaction.onsuccess = () => {
    console.log("数据写入成功")
  }

  // 事务失败时触发的事件
  transaction.onerror = () => {
    console.log("数据写入失败")
  }

  // 如果你还希望在添加请求成功时执行额外的逻辑，可以监听添加请求的成功事件
  addRequest.onsuccess = event => {
    console.log("数据添加请求成功", event.target.result)
    // event.target.result 包含了添加的数据的主键值
  }

  // 如果你还希望在添加请求失败时执行额外的逻辑，可以监听添加请求的失败事件
  addRequest.onerror = event => {
    console.log("数据添加请求失败", event.target.error)
  }
}

/**
 * 获取指定对象存储中的所有数据
 * @param {IDBDatabase} db - IndexedDB 数据库实例
 * @param {string} storeName - 对象存储的名称
 * @returns {Promise<Array<any>>} - 返回一个 Promise，解析为对象存储中的所有数据数组
 */
const getAllDataFromStore = (db, storeName) => {
  return new Promise((resolve, reject) => {
    // 启动只读事务
    const transaction = db.transaction([storeName], "readonly")

    // 获取对象存储进行操作
    const store = transaction.objectStore(storeName)

    const getAllRequest = store.getAll()
    // 获取所有数据成功，将数据数组返回
    getAllRequest.onsuccess = () => {
      console.log("成功")
      resolve(getAllRequest.result)
    }

    // 事务成功时触发的事件
    transaction.onsuccess = () => {
      // // 获取对象存储中的所有数据
      // 获取所有数据成功，将数据数组返回
      resolve(getAllRequest.result)
    }

    // 事务失败时触发的事件
    transaction.onerror = () => {
      console.log(`获取 ${storeName} 对象存储中的数据失败`)
      reject(`获取 ${storeName} 对象存储中的数据失败`)
    }
  })
}

export const DBUtils = {
  openDB,
  addData,
  getAllDataFromStore
}
